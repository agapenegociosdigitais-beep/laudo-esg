# Eureka Terra — Sincronizacao automatica de dados geoespaciais
# Instalar: crontab /root/eureka-terra/scripts/crontab.txt

# Auto-sync inteligente todo dia a meia-noite
0 0 * * * /usr/bin/docker exec eureka_backend python scripts/auto_sync.py >> /var/log/eureka_auto_sync.log 2>&1

# Rotacao de log (manter 30 dias)
0 1 * * * find /var/log/eureka_*.log -mtime +30 -delete
